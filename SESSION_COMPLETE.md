### 资源预加载 API

```typescript
// 添加预加载提示
await invoke('preloader_add_hint', {
  hint: {
    url: 'https://example.com/style.css',
    resource_type: 'Stylesheet',
    priority: 'High',
    source_url: 'https://example.com'
  }
});

// 处理预加载队列
const processed = await invoke('preloader_process_queue');

// 获取缓存资源
const data = await invoke('preloader_get_cached', {
  url: 'https://example.com/style.css'
});

// 学习访问模式
await invoke('preloader_learn_pattern', {
  source_url: 'https://example.com',
  resource_url: 'https://example.com/app.js'
});

// 获取预测提示
const hints = await invoke('preloader_get_predictive_hints', {
  url: 'https://example.com'
});

// 获取统计
const stats = await invoke('preloader_get_stats');
console.log(`缓存命中率: ${stats.hit_rate * 100}%`);
```
